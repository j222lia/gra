import pygame
import random
import math
 
# --- INICJALIZACJA PYGAME ---
pygame.init()
szerokosc_okna = 800
wysokosc_okna = 400
screen = pygame.display.set_mode((szerokosc_okna, wysokosc_okna))
pygame.display.set_caption("Gra z przeszkodami")
clock = pygame.time.Clock()                                                                #zegar do kontrolowania liczby klatek na sekundę
 
#czcionki
font_hud = pygame.font.Font(None, 40)
font_komunikat = pygame.font.Font(None, 70)
 
# --- KLASY OBIEKTÓW ---
class Kwadrat:
    def __init__(self): 
        self.bok = 40.0
        self.x = 100.0                                                                     #pozycja pozioma gracza (stała)
        self.y = 300.0 - self.bok                                                          
        self.predkosc_y = 0.0
        self.grawitacja = 0.6                                                              #przyspieszenie grawitacyjne co klatkę
        self.sila_skoku = -10.0                                                            #prędkość nadawana przy skoku (ujemna bo w pygame minus oznacza skok w górę)
        self.licznik_skokow = 0                                                            #ile razy gracz skoczył od ostatniego lądowania (pozwala na podwójny skok)
        self.kat_obrotu = 0.0
        self.predkosc_obrotu = 0.0
 
    def _oblicz_czas_lotu(self):                                                           #symuluje lot gracza, żeby policzyć ile klatek zajmie powrót na podłogę
        y = self.y                                                                         #kopia aktualnej pozycji y do symulacji
        v = self.predkosc_y                                                                #kopia aktualnej prędkości do symulacji
        n = 0                                                                              #licznik symulowanych klatek
        while y + self.bok < 300.0 and n < 1000:                                           #dopóki dół kwadratu nie dotarł do podłogi (i zabezpieczenie przed pętlą)
            v += self.grawitacja                                                           #symulowana grawitacja zwiększa prędkość
            y += v                                                                         #symulowana pozycja zmienia się o prędkość
            n += 1                                                                         #kolejna klatka
        return max(n, 1)                                                                   #zwraca liczbę klatek lotu 

    def skocz(self): 
        if self.licznik_skokow < 2:                                            
            self.predkosc_y = self.sila_skoku                                    
            self.licznik_skokow += 1                                             
            klatki_do_ladowania = self._oblicz_czas_lotu()                                 #policz ile klatek zajmie lądowanie
            self.predkosc_obrotu = 360.0 / klatki_do_ladowania                             #dzielenie przez czas lotu aby wyliczyć prędkość, która pozwoli wylądować na płasko
 
    def aktualizuj(self, wszystkie_platformy):                                             #dodawanie grawitacji co klatkę, żeby gracz powoli zaczął spadać
        self.predkosc_y += self.grawitacja                                         
        self.y += self.predkosc_y                                                    
 
        if self.licznik_skokow > 0:                                                   
            self.kat_obrotu += self.predkosc_obrotu                                       #obracaj kwadrat o wyliczoną prędkość obrotu
 
        # 1. Schody podnoszące poziom podłogi
        docelowa_podloga = 300.0                                                        
        srodek_gracza = self.x + self.bok / 2.0                                           #pozioma pozycja środka gracza (by wchodzenie po schodach wyglądało normalnie)
 
        for p in wszystkie_platformy:                                                     
            if srodek_gracza > p.x and srodek_gracza <= p.x + 30:                         #jeśli gracz jest nad 1 stopniem (każdy stopień ma 30px)
                docelowa_podloga = 270.0                                                  #podnieś podłogę na wysokość 1 stopnia
            elif srodek_gracza > p.x + 30 and srodek_gracza <= p.x + 60:                  
                docelowa_podloga = 240.0                                                          
            elif srodek_gracza > p.x + 60 and srodek_gracza <= p.x + 90:                          
                docelowa_podloga = 210.0                                                             
 
        # 2. Bezpieczne lądowanie
        if self.y + self.bok >= docelowa_podloga and self.predkosc_y >= 0:                #jeśli dół gracza dotarł do podłogi i gracz nie leci w górę
            self.y = docelowa_podloga - self.bok                                          #ustaw gracza dokładnie na podłodze
            self.predkosc_y = 0.0                                                         #zatrzymanie spadania
            self.licznik_skokow = 0                                                                        
            self.kat_obrotu = round(self.kat_obrotu / 90.0) * 90.0                        #zaokrąglij kąt obrotu do najbliższej wielokrotności 90°, by stał równo na podłodze
 
    def rysuj(self, ekran):                                                               #tworzymy powierzchnię kwadratu, żeby móc go obrócić
        surf = pygame.Surface((self.bok, self.bok), pygame.SRCALPHA)                           
        surf.fill((255, 50, 50))                                                                                
        rotated_surf = pygame.transform.rotate(surf, -self.kat_obrotu)                   #obróć powierzchnię o aktualny kąt (minus, by kręcić zgodnie z ruchem wskazówek zegara)
        rect = rotated_surf.get_rect(center=(self.x + self.bok / 2.0, self.y + self.bok / 2.0))     #środek obróconej figury pokrywa się ze środkiem gracza
        ekran.blit(rotated_surf, rect)                                                              #narysuj obróconą powierzchnię na ekranie
 
    def sprawdz_kolizje(self, przeszkoda): 
        margines_gracza = 6                                                              #zmniejszenie strefy zderzenia gracza o 6px z każdej strony 
        margines_przeszkody_bok = 12                                                     #zmniejszenie strefy zderzenia przeszkody w bokach o 12px
        margines_przeszkody_gora = 15                                                    #zmniejszenie strefy zderzenia przeszkody od góry o 15px
 
        gracz_lewo = self.x + margines_gracza                                            #lewa krawędź strefy zderzenia gracza
        gracz_prawo = self.x + self.bok - margines_gracza                                                               
        gracz_gora = self.y + margines_gracza                                                                            
        gracz_dol = self.y + self.bok - margines_gracza                                                                  
 
        przeszkoda_lewo = przeszkoda.x + margines_przeszkody_bok                         #lewa krawędź strefy zderzenia przeszkody
        przeszkoda_prawo = przeszkoda.x + przeszkoda.szerokosc - margines_przeszkody_bok                                    
        przeszkoda_gora = przeszkoda.y - przeszkoda.wysokosc + margines_przeszkody_gora                                      
        przeszkoda_dol = przeszkoda.y                                                                                        
 
        if (gracz_lewo < przeszkoda_prawo and                                            #test nakładania się gracza i przeszkody na siebie w pionie i poziomie
            gracz_prawo > przeszkoda_lewo and                                            
            gracz_gora < przeszkoda_dol and                                              
            gracz_dol > przeszkoda_gora):                                                
            return True                                                                  #jeśli wszystkie warunki spełnione - kolizja
        return False                                                                     #w przeciwnym razie brak kolizji

class Trojkat:                                                                           # zielony trójkąt na podłodze
    def __init__(self, start_x): 
        self.x = start_x                                                                 # pozycja pozioma (startuje przy prawej krawędzi ekranu)
        self.y = 300.0                                                                   # pozycja pionowa - stoi na podłodze
        self.szerokosc = 40.0                                                            # szerokość podstawy trójkąta
        self.wysokosc = 50.0                                                             # wysokość trójkąta

    def aktualizuj(self, predkosc): 
        self.x -= predkosc                                                               # przesuń trójkąt w lewo o konkretną prędkość (symulacja ruchu tła)

    def rysuj(self, ekran): 
        punkty = [
            (self.x, self.y),                                                           # lewy-dolny róg
            (self.x + self.szerokosc, self.y),                                          # prawy-dolny róg
            (self.x + self.szerokosc / 2.0, self.y - self.wysokosc)                     # górny czubek (środek, podniesiony o wysokość)
        ]
        pygame.draw.polygon(ekran, (50, 200, 50), punkty)                               # narysuj trójkąt w kolorze zielonym

class WysokiTrojkat:                                                                    # czerwony, wyższy trójkąt
    def __init__(self, start_x):
        self.x = start_x                                                                # pozycja pozioma startowa
        self.y = 300.0                                                                  # stoi na podłodze
        self.szerokosc = 50.0                                                           # szerokość podstawy
        self.wysokosc = 95.0                                                            # wysokość 

    def aktualizuj(self, predkosc):
        self.x -= predkosc                                                              # ruch w lewo

    def rysuj(self, ekran):
        punkty = [
            (self.x, self.y),                                                           # lewy-dolny róg
            (self.x + self.szerokosc, self.y),                                          # prawy-dolny róg
            (self.x + self.szerokosc / 2.0, self.y - self.wysokosc)                     # górny czubek
        ]
        pygame.draw.polygon(ekran, (200, 50, 50), punkty)                               # rysuj w kolorze czerwonym

class LatajacyTrojkat:                                                                  # trójkąt wiszącego w powietrzu, obracający się
    def __init__(self, start_x):
        self.x = start_x                                                                # pozycja pozioma startowa
        self.y = 220.0                                                                  # pozycja pionowa zawieszona w powietrzu
        self.szerokosc = 50.0                                                           # szerokość trójkąta
        self.wysokosc = 50.0                                                            # wysokość trójkąta
        self.kat = 0.0                                                                  # aktualny kąt obrotu 

    def aktualizuj(self, predkosc):
        self.x -= predkosc                                                              # ruch w lewo
        self.kat += 8.0                                                                 # co klatkę obróć trójkąt o 8°

    def rysuj(self, ekran):
        surf = pygame.Surface((self.szerokosc, self.wysokosc), pygame.SRCALPHA)                            # osobna przezroczysta powierzchnia (potrzebna do obrotu)
        punkty = [(0, self.wysokosc), (self.szerokosc, self.wysokosc), (self.szerokosc / 2, 0)]            # punkty trójkąta względem podłoża (0,0)
        pygame.draw.polygon(surf, (200, 50, 200), punkty)                                                  # narysowany trójkąt na tej powierzchni w kolorze fioletowym
        
        rotated_surf = pygame.transform.rotate(surf, -self.kat)                                            # obrót powierzchni o aktualny kąt
        rect = rotated_surf.get_rect(center=(self.x + self.szerokosc / 2.0, self.y - self.wysokosc / 2.0)) # ustawia obrócony trójkąt na właściwej pozycji
        ekran.blit(rotated_surf, rect)                                                                     # wyświetla obrócony trójkąt na ekranie

class Schody:                                                                          # klasa schodów, która podnosi poziom podłogi
    def __init__(self, start_x):
        self.x = start_x                                                               # pozycja pozioma startowa
        self.y = 300.0                                                                 # pozycja pionowa
        self.szerokosc = 90.0                                                          # szerokość schodów 
        self.wysokosc = 90.0                                                           # wysokość najwyższego stopnia

    def aktualizuj(self, predkosc):
        self.x -= predkosc                                                             # ruch w lewo

    def rysuj(self, ekran):
        pygame.draw.rect(ekran, (255, 140, 0), (self.x, self.y - 30, 30, 30))          # 1 stopień 
        pygame.draw.rect(ekran, (255, 140, 0), (self.x + 30, self.y - 60, 30, 60))     # 2 stopień
        pygame.draw.rect(ekran, (255, 140, 0), (self.x + 60, self.y - 90, 30, 90))     # 3 stopień

# --- ZMIENNE STANU GRY ---
gracz = Kwadrat()                                                                      # tworzy obiekt gracza
przeszkody = []                                                                        # lista trójkątów - dają punkty po ominięciu
platformy = []                                                                         # lista platform - nie dają punktów
punkty = 0                                                                             # licznik zdobytych punktów
stan_gry = "TRWA"                                                                      # aktualny stan gry: "trwa", "koniec_wygrana" lub "koniec_przegrana"
predkosc_przeszkod = 5.0                                                               # prędkość poruszania się przeszkód/schodów
kolor_tla = (180, 180, 180)                                                            # początkowy kolor tła (szary)

# --- GŁÓWNA PĘTLA GRY --- 
running = True                                                                         # gra działa, dopóki wartość jest True
while running:
    # 1. Obsługa zdarzeń i klawiszy
    for event in pygame.event.get():                                                  # sprwdza zdarzenia w grze (klawisze, zamknięcie okna itd.)
        if event.type == pygame.QUIT:                                                 # jeśli użytkownik zamknął okno (kliknął X)
            running = False                                                           # zakończ główną pętlę gry
        if event.type == pygame.KEYDOWN:                                              # jeśli wciśnięto jakiś klawisz
            if event.key == pygame.K_SPACE and stan_gry == "TRWA":                    # jeśli to spacja i gra wciąż trwa
                gracz.skocz()                                                         # wykonuje skok
                
    # 2. Zmiana poziomu trudności
    if punkty >= 30:                                                                  # jeśli gracz zdobył 30+ punktów
        stan_gry = "KONIEC_WYGRANA"                                                   # gra kończy się wygraną
    elif punkty >= 20:                                                                # od 20 do 29 punktów
        kolor_tla = (138, 43, 226)                                                    # fioletowe tło
        predkosc_przeszkod = 10.0                                                     # najwyższa prędkość przeszkód
    elif punkty >= 10:                                                                # od 10 do 19 punktów
        kolor_tla = (70, 130, 180)                                                    # niebieskie tło
        predkosc_przeszkod = 7.5                                                      # średnia prędkość przeszkód
    else:                                                                             # poniżej 10 punktów
        kolor_tla = (180, 180, 180)                                                   # domyślne szare tło
        predkosc_przeszkod = 5.0                                                      # podstawowa prędkość
         
    # 3. Rysowanie tła i podłogi
    screen.fill(kolor_tla)                                                           # wypełnia ekran kolorem tła
    pygame.draw.line(screen, (0, 0, 0), (0, 300), (szerokosc_okna, 300), 3)          # narysuj czarną linię na y=300 (podłoga)
     
    # 4. Aktualizacja gry
    if stan_gry == "TRWA":                                                           # działa tylko gdy gra trwa
        gracz.aktualizuj(platformy)                                                  # aktualizuje pozycje gracza
         
        ostatni_x = 0                                                                # pozycja ostatniego obiektu
        if len(przeszkody) > 0:                                                      # jeśli są przeszkody
            ostatni_x = max(ostatni_x, przeszkody[-1].x)                             # pobiera ich pozycje
        if len(platformy) > 0:                                                       # jeśli są platformy
            ostatni_x = max(ostatni_x, platformy[-1].x)                              # pobiera ich pozycje
             
        # Losowanie przeszkód
        if ostatni_x == 0 or ostatni_x < szerokosc_okna - 450:                            # jeśli nie ma jeszcze obiektów, albo jest dość miejsca (min. 450px odstępu)
            los = random.randint(1, 100)                                                  # wylosuj liczbę od 1 do 100, decydującą co się pojawi
            if punkty >= 20:                                                              # przy 20+ punktach dostępne są wszystkie typy przeszkód
                if los < 30: przeszkody.append(LatajacyTrojkat(szerokosc_okna))           # 29% szans: latający trójkąt
                elif los < 50: przeszkody.append(WysokiTrojkat(szerokosc_okna))           # 20% szans: wysoki trójkąt
                elif los < 70: platformy.append(Schody(szerokosc_okna))                   # 20% szans: schody
                else: przeszkody.append(Trojkat(szerokosc_okna))                          # 30% szans: zwykły trójkąt
            elif punkty >= 10:                                                            # przy 10-19 punktach - bez latającego trójkąta
                if los < 40: przeszkody.append(WysokiTrojkat(szerokosc_okna))             # 39% szans: wysoki trójkąt
                elif los < 70: platformy.append(Schody(szerokosc_okna))                   # 30% szans: schody
                else: przeszkody.append(Trojkat(szerokosc_okna))                          # 30% szans: zwykły trójkąt
            else:                                                                         # poniżej 10 punktów - tylko najłatwiejsze przeszkody
                if los < 30: platformy.append(Schody(szerokosc_okna))                     # 29% szans: schody
                else: przeszkody.append(Trojkat(szerokosc_okna))                          # 70% szans: zwykły trójkąt
                 
        # Platformy znikają, ALE NIE DAJĄ PUNKTÓW
        for i in range(len(platformy) - 1, -1, -1):                                       # iteracja od końca listy (bezpieczne usuwanie podczas iteracji)
            p = platformy[i]                                                              # pobierz kolejną platformę
            p.aktualizuj(predkosc_przeszkod)                                              # przesuń ją w lewo
            if p.x + p.szerokosc < 0:                                                     # jeśli całkowicie zniknęła z lewej strony ekranu
                platformy.pop(i)                                                          # usuń ją z listy (bez dodawania punktów)

        # Trójkąty znikają i DAJĄ PUNKTY
        for i in range(len(przeszkody) - 1, -1, -1):                                      # iteracja od końca listy przeszkód
            p = przeszkody[i]                                                             # pobierz kolejną przeszkodę
            p.aktualizuj(predkosc_przeszkod)                                              # przesuń ją w lewo
            if p.x + p.szerokosc < 0:                                                     # jeśli zniknęła z ekranu (gracz ją ominął)
                przeszkody.pop(i)                                                         # usuń ją z listy
                punkty += 1                                                               # i dodaj punkt za jej ominięcie
            elif gracz.sprawdz_kolizje(p):                                                # w przeciwnym razie sprawdź, czy gracz w nią uderzył
                stan_gry = "KONIEC_PRZEGRANA"                                             # jeśli tak - gra kończy się przegraną
                 
    # 5. Wyświetlanie obiektów na ekranie
    gracz.rysuj(screen)                                                                   # narysuj gracza
    for p in platformy:                                                                   # dla każdej platformy (schody)
        p.rysuj(screen)                                                                   # narysuj ją
    for p in przeszkody:                                                                  # dla każdej przeszkody (trójkąt)
        p.rysuj(screen)                                                                   # narysuj ją
         
    # 6. Teksty (punkty i koniec gry)
    tekst_punkty = font_hud.render("Punkty: " + str(punkty), True, (255, 255, 255))                       # tworzy napis z punktami 
    screen.blit(tekst_punkty, (10, 10))                                                                   # narysuj go w lewym górnym rogu ekranu
     
    if stan_gry == "KONIEC_WYGRANA":                                                                      # jeśli gra zakończyła się wygraną
        tekst_wygrana = font_komunikat.render("GRATULACJE!", True, (255, 215, 0))                         # przygotuj złoty napis "GRATULACJE!"
        rect_wygrana = tekst_wygrana.get_rect(center=(szerokosc_okna / 2, wysokosc_okna / 2 - 50))        # wyśrodkuj go 
        screen.blit(tekst_wygrana, rect_wygrana)                                                          # narysuj napis na ekranie
    elif stan_gry == "KONIEC_PRZEGRANA":                                                                  # jeśli gra zakończyła się przegraną
        tekst_przegrana = font_komunikat.render("KONIEC GRY!", True, (255, 0, 0))                         # przygotuj czerwony napis "KONIEC GRY!"
        rect_przegrana = tekst_przegrana.get_rect(center=(szerokosc_okna / 2, wysokosc_okna / 2 - 50))    # wyśrodkuj go
        screen.blit(tekst_przegrana, rect_przegrana)                                                      # narysuj napis na ekranie

    # Odświeżenie okna i kontrola klatek (60 FPS)
    pygame.display.flip()                                                                  # pokaż na ekranie to, co zostało narysowane w tej klatce
    clock.tick(60)                                                                         # utrzymuj 60 FPS

pygame.quit()                                                                              # po zakończeniu pętli zamyka pygame
